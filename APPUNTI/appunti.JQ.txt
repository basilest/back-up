                              jq    ....  "secure-mongo-db1.live.aws.internal"

    jq '.[] | select(.location=="Stockholm")'  filter only entries with a field equal a value


    jq '.[] | .login, .id'   to select only the fields I want (here 2 fields "login" "id")
